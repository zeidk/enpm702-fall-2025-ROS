ros702() {
    # --- ROS 2 and Workspace Setup ---
    local WS="$HOME/Documents/Teaching/fall2025/ENPM702/ros2_ws"
    local ROS2_DISTRO="jazzy"

    # Check if ROS installation exists
    if [ -f "/opt/ros/${ROS2_DISTRO}/setup.zsh" ]; then
        source "/opt/ros/${ROS2_DISTRO}/setup.zsh"
    else
        echo "‚ö†Ô∏è ROS 2 distro '${ROS2_DISTRO}' not found at /opt/ros"
        return 1
    fi

    # Source workspace if built
    if [ -f "${WS}/install/setup.zsh" ]; then
        source "${WS}/install/setup.zsh"
        echo "üì¶ Workspace overlay loaded: ${WS}"
    else
        echo "‚ö†Ô∏è Workspace not built yet: ${WS}/install/setup.zsh missing"
    fi

    # Setup argcomplete (only once per shell session)
    if command -v register-python-argcomplete >/dev/null; then
        eval "$(register-python-argcomplete ros2)"
        eval "$(register-python-argcomplete colcon)"
    fi

    cd "${WS}" || return
    echo "‚û°Ô∏è Current directory: ${WS}"
}
